14pub fn get_f64_value(value: &Option<impl F64Compatible>) -> Result<f64, SignalKGetError> {
15    if let Some(ref number_value) = value {
16        if let Some(f64_value) = number_value.as_f64() {
17            Ok(f64_value)
18        } else {
19            Err(SignalKGetError::ValueNotSet)
20        }
21    } else {
22        Err(SignalKGetError::ValueNotSet)

26pub trait Path<T> {
27    fn get_path(&self, path: &[&str]) -> Result<T, SignalKGetError>;
28}
29
30pub fn get_path<T>(path: &[&str], value: &Option<&impl Path<T>>) -> Result<T, SignalKGetError> {
31    debug!("get_path({:?}, ...)", path);
32    if let Some((_start, elements)) = path.split_first() {
33        if let Some(item) = value {
34            item.get_path(elements)
35        } else {
36            warn!("Value not set for path {:?}", path);
37            Err(SignalKGetError::ValueNotSet)
38        }
39    } else {
40        Err(SignalKGetError::NoSuchPath)
41    }
42}

51) -> Result<f64, SignalKGetError> {
52    debug!("get_f64_value_for_path({:?}, ...)", path);
53    if let Some(ref item) = value {
54        path.remove(0);
55        item.get_f64_for_path(path)
56    } else {
57        Err(SignalKGetError::ValueNotSet)
58    }
