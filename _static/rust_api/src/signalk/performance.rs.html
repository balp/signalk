29impl F64CompatiblePath for V1Performance {
30    fn get_f64_for_path(&self, path: &mut Vec<&str>) -> Result<f64, SignalKGetError> {
31        match path[0] {
32            "polars" => Err(SignalKGetError::WrongDataType),
33            "activePolar" => Err(SignalKGetError::WrongDataType),
34            "activePolarData" => Err(SignalKGetError::WrongDataType),
35            "polarSpeed" => get_f64_value(&self.polar_speed),
36            "polarSpeedRatio" => get_f64_value(&self.polar_speed_ratio),
37            "velocityMadeGood" => get_f64_value(&self.velocity_made_good),
38            "velocityMadeGoodToWaypoint" => get_f64_value(&self.velocity_made_good_to_waypoint),
39            "beatAngle" => get_f64_value(&self.beat_angle),
40            "beatAngleVelocityMadeGood" => get_f64_value(&self.beat_angle_velocity_made_good),
41            "beatAngleTargetSpeed" => get_f64_value(&self.beat_angle_target_speed),
42            "gybeAngle" => get_f64_value(&self.gybe_angle),
43            "gybeAngleVelocityMadeGood" => get_f64_value(&self.gybe_angle_velocity_made_good),
44            "gybeAngleTargetSpeed" => get_f64_value(&self.gybe_angle_target_speed),
45            "targetAngle" => get_f64_value(&self.target_angle),
46            "targetSpeed" => get_f64_value(&self.target_speed),
47            "leeway" => get_f64_value(&self.leeway),
48            "tackMagnetic" => get_f64_value(&self.tack_magnetic),
49            "tackTrue" => get_f64_value(&self.tack_true),
50            &_ => Err(SignalKGetError::NoSuchPath),
51        }
52    }

55impl V1Performance {
56    pub fn update(&mut self, path: &mut Vec<&str>, value: &serde_json::value::Value) {
57        if path.is_empty() {
58            return;
59        }
60        match path[0] {
61            // "polars" => self.polars = V1NumberValue::from_value(value),
62            "activePolar" => self.active_polar = V1StringValue::from_value(value),
63            // "activePolarData" => self.active_polar_data = V1NumberValue::from_value(value),
64            "polarSpeed" => self.polar_speed = V1NumberValue::from_value(value),
65            "polarSpeedRatio" => self.polar_speed_ratio = V1NumberValue::from_value(value),
66            "velocityMadeGood" => self.velocity_made_good = V1NumberValue::from_value(value),
67            "velocityMadeGoodToWaypoint" => {
68                self.velocity_made_good_to_waypoint = V1NumberValue::from_value(value)
69            }
70            "beatAngle" => self.beat_angle = V1NumberValue::from_value(value),
71            "beatAngleVelocityMadeGood" => {
72                self.beat_angle_velocity_made_good = V1NumberValue::from_value(value)
73            }
74            "beatAngleTargetSpeed" => {
75                self.beat_angle_target_speed = V1NumberValue::from_value(value)
76            }
77            "gybeAngle" => self.gybe_angle = V1NumberValue::from_value(value),
78            "gybeAngleVelocityMadeGood" => {
79                self.gybe_angle_velocity_made_good = V1NumberValue::from_value(value)
80            }
81            "gybeAngleTargetSpeed" => {
82                self.gybe_angle_target_speed = V1NumberValue::from_value(value)
83            }
84            "targetAngle" => self.target_angle = V1NumberValue::from_value(value),
85            "targetSpeed" => self.target_speed = V1NumberValue::from_value(value),
86            "leeway" => self.leeway = V1NumberValue::from_value(value),
87            "tackMagnetic" => self.tack_magnetic = V1NumberValue::from_value(value),
88            "tackTrue" => self.tack_true = V1NumberValue::from_value(value),
89
90            &_ => {
91                log::warn!(
92                    "V1Performance: Unknown update pattern: {:?}::{:?}",
93                    path,
94                    value
95                );
96            }
97        }
98    }

121            }
