8/// These items can be updated by a V1UpdateType
9///
10/// Given the GetContext Trait one can fine a matching object
11/// to apply the update into from the V1DeltaFormat's context
12pub trait Updatable {
13    /// This will change the found object to have a a new value
14    fn apply_update(&mut self, update: &V1UpdateType);
15
16    /// This will return the ID if possible for the object
17    ///
18    /// At the moment this is mainly to make sure we can easily
19    /// verify the GetContext functionality
20    fn id(&self) -> String;
21
22    /// This will return the "type" of the context
23    ///
24    /// At the moment this is mainly to make sure we can easily
25    /// verify the GetContext functionality
26    fn type_name(&self) -> String;
27}

83    pub fn apply_delta(&mut self, delta: &V1DeltaFormat) {
84        if let Some(ref context) = delta.context {
85            let v: Vec<&str> = context.split('.').collect();
86            if v.len() < 2 {
87                return;
88            }
89            if v[0] == "vessels" {
90                if self.vessels.is_none() {
91                    self.vessels = Some(HashMap::new());
92                }
93                if let Some(ref mut vessels) = self.vessels {
94                    let id = v[1].to_string();
95                    if !vessels.contains_key(&id) {
96                        vessels.insert(id.clone(), V1Vessel::new_with_id(&id));
97                    }
98                    let mut t = vessels.get_mut(&id);
99                    if let Some(ref mut vessel) = t {
100                        for update in &delta.updates {
101                            vessel.apply_update(update);
102                        }
103                    }
104                }
105            }
106        }
107    }
108
109    pub fn get_f64_for_path(&self, path: String) -> Result<f64, SignalKGetError> {
110        let mut path_que: Vec<&str> = path.split('.').collect();
111        log::debug!("get_f64_for_path(&self, {:?})", path_que);
112        match path_que[0] {
113            "version" => Err(SignalKGetError::WrongDataType),
114            "self" => {
115                let self_path: Vec<&str> = self.self_.split('.').collect();
116                if self_path[0] == "vessels" {
117                    if let Some(ref vessels) = self.vessels {
118                        if let Some(vessel) = vessels.get(self_path[1]) {
119                            path_que.remove(0);
120                            vessel.get_f64_for_path(&mut path_que)
121                        } else {
122                            Err(SignalKGetError::NoSuchPath)
123                        }
124                    } else {
125                        Err(SignalKGetError::NoSuchPath)
126                    }
127                } else {
128                    Err(SignalKGetError::NoSuchPath)
129                }
130            }
131            "vessels" => {
132                if let Some(ref vessels) = self.vessels {
133                    if let Some(vessel) = vessels.get(path_que[1]) {
134                        path_que.remove(0);
135                        path_que.remove(0);
136                        vessel.get_f64_for_path(&mut path_que)
137                    } else {
138                        Err(SignalKGetError::NoSuchPath)
139                    }
140                } else {
141                    Err(SignalKGetError::NoSuchPath)
142                }
143            }
144            &_ => Err(SignalKGetError::NoSuchPath),
145        }
146    }
147
148    pub fn get_self(&self) -> Option<&V1Vessel> {
149        if let Some(vessels) = self.vessels.as_ref() {
150            let v: Vec<&str> = self.self_.split('.').collect();
151            if v.len() >= 2 {
152                return vessels.get(v[1]);
153            } else {
154                return vessels.get(v[0]);
155            }
156        };
157        None
158    }

161/// Builder for the Signal K Full format structure
162pub struct V1FullFormatBuilder {
163    version: String,
164    self_: String,
165    vessels: Option<HashMap<String, V1Vessel>>,
166    sources: Option<V1Sources>,
167}
168
169impl Default for V1FullFormatBuilder {
170    fn default() -> Self {
171        V1FullFormatBuilder {
172            version: "1.7.0".to_string(),
173            self_: "".to_string(),
174            vessels: None,
175            sources: None,
176        }
177    }
178}
179
180impl V1FullFormatBuilder {
181    pub fn version(mut self, version: String) -> V1FullFormatBuilder {
182        self.version = version;
183        self
184    }
185    pub fn self_(mut self, self_: String) -> V1FullFormatBuilder {
186        self.self_ = self_;
187        self
188    }
189    pub fn add_vessel(mut self, key: String, vessel: V1Vessel) -> V1FullFormatBuilder {
190        if self.vessels.is_none() {
191            self.vessels = Some(HashMap::new());
192        }
193        if let Some(ref mut x) = self.vessels {
194            x.insert(key, vessel);
195        }
196        self
197    }
198    pub fn sources(mut self, sources: V1Sources) -> V1FullFormatBuilder {
199        self.sources = Some(sources);
200        self
201    }
202    pub fn build(self) -> V1FullFormat {
203        V1FullFormat {
204            version: self.version,
205            self_: self.self_,
206            vessels: self.vessels,
207            // sources: self.sources,
208        }
209    }
210}
